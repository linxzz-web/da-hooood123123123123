-- CONFIGURATION
local TARGET_USERNAME = "zdenda3030"
local RISE_SPEED = 2
local UNDER_OFFSET = 3
local ITEM_NAME = "[WALLET]"

-- Services
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer
local UserInputService = game:GetService("UserInputService")

-- Wait for player
repeat task.wait() until LocalPlayer and LocalPlayer.Character
print("‚úÖ Player loaded: " .. LocalPlayer.Name)

-- Find target
local targetPlayer = nil
for _, player in ipairs(Players:GetPlayers()) do
    if player.Name:lower() == TARGET_USERNAME:lower() then
        targetPlayer = player
        break
    end
end

if targetPlayer then
    print("‚úÖ Target found: " .. targetPlayer.Name)
else
    print("‚ùå Target NOT found: " .. TARGET_USERNAME)
end

-- Simple item check function
local function checkForWallet()
    print("\n--- CHECKING FOR WALLET ---")
    local found = false
    
    -- Check Backpack
    local backpack = LocalPlayer:FindFirstChild("Backpack")
    if backpack then
        print("Backpack items:")
        for _, item in ipairs(backpack:GetChildren()) do
            print("  - " .. item.Name)
            if item.Name == ITEM_NAME then
                print("  ‚úÖ FOUND IN BACKPACK!")
                found = true
            end
        end
    else
        print("No Backpack found")
    end
    
    -- Check Character
    if LocalPlayer.Character then
        print("Character items:")
        for _, item in ipairs(LocalPlayer.Character:GetChildren()) do
            if item:IsA("Tool") then
                print("  - " .. item.Name)
                if item.Name == ITEM_NAME then
                    print("  ‚úÖ FOUND ON CHARACTER!")
                    found = true
                end
            end
        end
    end
    
    print("Wallet found: " .. tostring(found))
    print("------------------------\n")
    return found
end

-- Manual rising function
local rising = false
local connection = nil

local function startRising()
    if rising then 
        print("Already rising")
        return
    end
    if not targetPlayer or not targetPlayer.Character then
        print("Target not available")
        return
    end
    
    print("üîº STARTING RISE")
    rising = true
    
    connection = RunService.Heartbeat:Connect(function()
        if not rising then return end
        if not targetPlayer or not targetPlayer.Character then return end
        if not LocalPlayer.Character then return end
        
        local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        local localRoot = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        
        if targetRoot and localRoot then
            local targetY = targetRoot.Position.Y - UNDER_OFFSET
            local currentY = localRoot.Position.Y
            
            if currentY < targetY then
                local newY = currentY + (RISE_SPEED * 0.1)
                if newY > targetY then newY = targetY end
                localRoot.CFrame = CFrame.new(localRoot.Position.X, newY, localRoot.Position.Z)
            end
        end
    end)
end

local function stopRising()
    if not rising then return end
    print("üîΩ STOPPING RISE")
    rising = false
    if connection then
        connection:Disconnect()
        connection = nil
    end
end

-- Key binds
UserInputService.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.I then
        -- I = Check inventory
        checkForWallet()
        
    elseif input.KeyCode == Enum.KeyCode.R then
        -- R = Start rising
        startRising()
        
    elseif input.KeyCode == Enum.KeyCode.S then
        -- S = Stop rising
        stopRising()
        
    elseif input.KeyCode == Enum.KeyCode.T then
        -- T = Toggle
        if rising then
            stopRising()
        else
            startRising()
        end
        
    elseif input.KeyCode == Enum.KeyCode.H then
        -- H = Help
        print("\n=== CONTROLS ===")
        print("I - Check for wallet")
        print("R - Start rising")
        print("S - Stop rising")
        print("T - Toggle rise")
        print("H - This help")
        print("================\n")
    end
end)

print("\n‚úÖ SIMPLE DEBUG SCRIPT LOADED")
print("Target: " .. TARGET_USERNAME)
print("Press H for help")
print("Press I to check for wallet\n")
